Create `molecule/default/converge.yml`:
```yaml
---
- name: Converge
  hosts: all
  become: true
  gather_facts: true

  tasks:
    - name: Include usbguard role
      ansible.builtin.include_role:
        name: malpanez.security.usbguard
      vars:
        usbguard_enabled: true
        security_mode: enforce
```

The converge will hit `meta: end_host` in enforce.yml because Docker containers have no USB bus at `/sys/bus/usb/devices`. This is expected behavior — converge completes with rc=0.

Create `molecule/default/verify.yml`:
```yaml
---
- name: Verify usbguard
  hosts: all
  become: true
  gather_facts: true

  tasks:
    - name: Check USB bus availability in container
      ansible.builtin.stat:
        path: /sys/bus/usb/devices
      register: _verify_usb_bus

    - name: Assert USB bus absent in container environment
      ansible.builtin.assert:
        that:
          - not _verify_usb_bus.stat.exists
        fail_msg: >-
          USB bus was unexpectedly found at /sys/bus/usb/devices.
          Review container detection logic.
        quiet: true

    - name: Gather package facts
      ansible.builtin.package_facts:
        manager: auto

    - name: Assert usbguard package is NOT installed (enforce skipped in container)
      ansible.builtin.assert:
        that:
          - "'usbguard' not in ansible_facts.packages"
        fail_msg: >-
          usbguard package was installed despite USB bus being absent.
          meta:end_host should have prevented enforcement.
        quiet: true

    - name: Confirm graceful container skip
      ansible.builtin.debug:
        msg: >-
          Container verification complete.
          USB bus absent, enforce skipped gracefully via meta:end_host.
          No enforcement errors occurred.
```

The verify.yml asserts:
1. USB bus is absent (confirms container detection works)
2. usbguard package is NOT installed (confirms meta:end_host fired before install tasks)
3. Debug message confirms graceful skip

Do NOT assert usbguard service is running — containers have no USB bus and enforce gracefully skips.
