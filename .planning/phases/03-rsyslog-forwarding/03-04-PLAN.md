**verify.yml** — assert key outcomes with container-aware guards. Following the ntp_hardening verify.yml pattern:

```yaml
- name: Verify rsyslog_forwarding
  hosts: all
  become: true
  gather_facts: true

  tasks:
    - name: Detect container environment
      ansible.builtin.command:
        cmd: systemd-detect-virt --container
      register: _virt_check
      changed_when: false
      failed_when: false

    - name: Set container flag
      ansible.builtin.set_fact:
        _in_container: "{{ _virt_check.rc == 0 }}"

    - name: Verify rsyslog package installed
      ansible.builtin.package_facts:
        manager: auto

    - name: Assert rsyslog is installed
      ansible.builtin.assert:
        that:
          - "'rsyslog' in ansible_facts.packages"
        fail_msg: rsyslog package is not installed
        quiet: true

    - name: Check forwarding drop-in exists
      ansible.builtin.stat:
        path: /etc/rsyslog.d/99-forwarding.conf
      register: _dropin_stat

    - name: Assert forwarding drop-in exists
      ansible.builtin.assert:
        that:
          - _dropin_stat.stat.exists
        fail_msg: /etc/rsyslog.d/99-forwarding.conf does not exist
        quiet: true

    - name: Assert drop-in file mode
      ansible.builtin.assert:
        that:
          - _dropin_stat.stat.mode == '0644'
        fail_msg: "99-forwarding.conf mode is {{ _dropin_stat.stat.mode }}, expected 0644"
        quiet: true

    - name: Check drop-in contains action(type="omfwd")
      ansible.builtin.command:
        cmd: grep 'action(type="omfwd"' /etc/rsyslog.d/99-forwarding.conf
      register: _dropin_content_check
      changed_when: false

    - name: Assert drop-in uses RainerScript syntax
      ansible.builtin.assert:
        that:
          - _dropin_content_check.rc == 0
        fail_msg: 99-forwarding.conf does not contain RainerScript action() syntax
        quiet: true

    - name: Check drop-in contains target host
      ansible.builtin.command:
        cmd: grep 'Target="127.0.0.1"' /etc/rsyslog.d/99-forwarding.conf
      register: _dropin_target_check
      changed_when: false

    - name: Assert drop-in targets 127.0.0.1
      ansible.builtin.assert:
        that:
          - _dropin_target_check.rc == 0
        fail_msg: 99-forwarding.conf does not contain Target="127.0.0.1"
        quiet: true

    - name: Run rsyslogd syntax check (non-container only)
      ansible.builtin.command:
        cmd: rsyslogd -N1 -f /etc/rsyslog.d/99-forwarding.conf
      register: _syntax_check
      changed_when: false
      failed_when: false
      when: not _in_container | bool

    - name: Assert rsyslogd syntax check passed
      ansible.builtin.assert:
        that:
          - _syntax_check.rc == 0
        fail_msg: rsyslogd -N1 syntax check failed
        quiet: true
      when: not _in_container | bool

    - name: Check rsyslog service enabled (non-container only)
      ansible.builtin.command:
        cmd: systemctl is-enabled rsyslog
      register: _svc_enabled
      changed_when: false
      failed_when: false
      when: not _in_container | bool

    - name: Assert rsyslog service enabled
      ansible.builtin.assert:
        that:
          - _svc_enabled.stdout == 'enabled'
        fail_msg: rsyslog service is not enabled
        quiet: true
      when: not _in_container | bool

    - name: Check rsyslog service active (non-container only)
      ansible.builtin.command:
        cmd: systemctl is-active rsyslog
      register: _svc_active
      changed_when: false
      failed_when: false
      when: not _in_container | bool

    - name: Assert rsyslog service active
      ansible.builtin.assert:
        that:
          - _svc_active.stdout == 'active'
        fail_msg: rsyslog service is not active
        quiet: true
      when: not _in_container | bool
```

All task names start uppercase. All modules use FQCN. All assert tasks have `fail_msg` and `quiet: true`.
